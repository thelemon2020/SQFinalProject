147  public void AddBalance(Contract contract)
148  {
149  if(contract.Cost > 0.0) // make sure a basic cost has been calculated for the contract first
150  {
151  if (contract.VanType == 0) // the van type is a Dry Van, no upcharge associated
152  {
153  if (contract.Quantity == 0) // the trip is FTL
154  {
155  double tmp = AddBalance(contract.Rate, contract.Distance);
156  Balance += tmp * contract.FTLUpCharge;
157  }
158  else //the trip is LTL
159  {
160  double tmp = AddBalance(contract.Rate, contract.Distance, contract.Quantity);
161  Balance += tmp * contract.LTLUpCharge;
162  }
163  }
164  else // The van type is a Reefer Van there is an associate upcharge
165  {
166  if (contract.Quantity == 0) // FTL in a Reefer van
167  {
168  double tmpBal = AddBalance(contract.Rate, contract.Distance);
169  tmpBal *= contract.ReeferUpCharge;
170  tmpBal *= contract.FTLUpCharge;
171  Balance += tmpBal;
172  }
173  else // LTL in a Reefer van
174  {
175  double tmpBal = AddBalance(contract.Rate, contract.Distance, contract.Quantity);
176  tmpBal *= contract.ReeferUpCharge;
177  tmpBal *= contract.LTLUpCharge;
178  Balance += tmpBal;
179  }
180  }
181  }
182  else // flat cost not assessed yet, add the contract to the UncalculatedContracts dictionary to be assessed at a later time.
183  {
184  UncalculatedContracts.Add(contract.ID, contract);
185  }
186  }
